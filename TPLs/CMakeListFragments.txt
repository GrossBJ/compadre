This file contains fragments of CMake code needed to supply GSL using a .cmake approach

FindPackageHandleStandardArgs.cmake contains APPEND on lists which requires CMake 3.4

cmake_minimum_required(VERSION 3.4)

SET(CMAKE_MODULE_PATH ${COMPADRE_MODULE_DIR} ".")
IF (DEFINED $ENV{GSL_ROOT_DIR})
  SET(CMAKE_PREFIX_PATH $ENV{GSL_ROOT_DIR} ${CMAKE_PREFIX_PATH})
  MESSAGE("GSL Found At: $ENV{GSL_ROOT_DIR}")
ENDIF()
FIND_PACKAGE(GSL REQUIRED) # REQUIRED tag can be removed after finished finding this
	SET(CMAKE_CXX_FLAGS "-lgsl -lcblas")
