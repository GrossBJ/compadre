676  norm_t_normal += T(_dimensions-1,0)*T(_dimensions-1,0);
677  T(_dimensions-1,1) = T(0,0) - T(1,0);
678  norm_t_normal += T(_dimensions-1,1)*T(_dimensions-1,1);
679  }
680  norm_t_normal = std::sqrt(norm_t_normal);
681  for (int i=0; i<_dimensions-1; ++i) {
682  T(_dimensions-1,i) /= norm_t_normal;
683  }
684  });

1003  */
1004

1007  Kokkos::single(Kokkos::PerThread(teamMember), [&] () {
1008  _prestencil_weights(0,0,0,0,0) = -1;

1010  });
1011  });
1012  } else if (_data_sampling_functional == ManifoldVectorPointSample) {

1015  for (int j=0; j<_dimensions; ++j) {
1016  for (int k=0; k<_dimensions-1; ++k) {
1017  _prestencil_weights(0,target_index,0,k,j) = T(k,j);
1018  }
1019  }

1021  });
1022  } else if (_data_sampling_functional == StaggeredEdgeIntegralSample) {
1023  compadre_kernel_assert_debug(_dense_solver_type==DenseSolverType::MANIFOLD && "StaggeredEdgeIntegralSample prestencil weight only written for manifolds.");

1025  Kokkos::parallel_for(Kokkos::TeamThreadRange(teamMember,this->getNNeighbors(target_index)), [=] (const int m) {

1027  for (int quadrature = 0; quadrature<_number_of_quadrature_points; ++quadrature) {
1028  XYZ tangent_quadrature_coord_2d;
1029  for (int j=0; j<_dimensions-1; ++j) {

1040  _prestencil_weights(1,target_index,m,0,j) += _parameterized_quadrature_sites[quadrature]*tangent_vector[j]*_quadrature_weights[quadrature];
1041  }
1042  }
1043  });
1044  });
1045  } else if (_data_sampling_functional == VaryingManifoldVectorPointSample) {
1046 
1047  scratch_vector_type delta(teamMember.thread_scratch(_scratch_thread_level_b), manifold_NP);
1048 
1049  Kokkos::parallel_for(Kokkos::TeamThreadRange(teamMember,this->getNNeighbors(target_index)), [&] (const int m) {
1050 
1051  Kokkos::single(Kokkos::PerThread(teamMember), [&] () {

1053  });

1055  double grad_xi1 = 0, grad_xi2 = 0;
1056  Kokkos::parallel_reduce(Kokkos::ThreadVectorRange(teamMember,this->getNNeighbors(target_index)), [=] (const int i, double &t_grad_xi1) {

1060  }

1062  double normal_coordinate = rel_coord[_dimensions-1];
1063 
1064  // apply coefficients to sample data
1065  t_grad_xi1 += alpha_ij * normal_coordinate;
1066  }, grad_xi1);
1067  t1(m) = grad_xi1;
1068 
1069  Kokkos::single(Kokkos::PerThread(teamMember), [&] () {
1070  this->calcGradientPij(delta.data(), target_index, m, 0 /*alpha*/, 1 /*partial_direction*/, _dimensions-1, _curvature_poly_order, false /*specific order only*/, &T, ReconstructionSpace::ScalarTaylorPolynomial, PointSample);
1071  });
1072  Kokkos::parallel_reduce(Kokkos::ThreadVectorRange(teamMember,this->getNNeighbors(target_index)), [=] (const int i, double &t_grad_xi2) {
1073  double alpha_ij = 0;
1074  for (int l=0; l<manifold_NP; ++l) {
1075  alpha_ij += delta(l)*Q(i,l);
1076  }

1078  double normal_coordinate = rel_coord[_dimensions-1];
1079 
1080  // apply coefficients to sample data

1082  }, grad_xi2);
1083  t2(m) = grad_xi2;
1084  });
1085 
1086  teamMember.team_barrier();
1087

1089 
1090  Kokkos::parallel_for(Kokkos::TeamThreadRange(teamMember,this->getNNeighbors(target_index)), [=] (const int m) {
1091  // constructs tangent vector at neighbor site
1092  Kokkos::single(Kokkos::PerThread(teamMember), [&] () {
1093  for (int j=0; j<_dimensions; ++j) {

1096  }
1097

1099  double norm = 0;
1100  for (int j=0; j<_dimensions; ++j) {
1101  norm += tangent(0,j)*tangent(0,j);
1102  }
1103

1106  for (int j=0; j<_dimensions; ++j) {
1107  tangent(0,j) /= norm;
1108  }
1109

1113  for (int j=0; j<_dimensions; ++j) {

1116  // normalize second vector
1117  norm = 0;
1118  for (int j=0; j<_dimensions; ++j) {
1119  norm += tangent(1,j)*tangent(1,j);
1120  }
1121  norm = std::sqrt(norm);
1122  for (int j=0; j<_dimensions; ++j) {
1123  tangent(1,j) /= norm;
1124  }
1125  }
1126 
1127  // stores matrix of tangent and normal directions as a prestencil weight
1128  for (int j=0; j<_dimensions; ++j) {
1129  for (int k=0; k<_dimensions-1; ++k) {
1130  _prestencil_weights(0,target_index,m,k,j) = tangent(k,j);
1131  }
1132  }
1133  });
1134  });
1135  }
1136  teamMember.team_barrier();
1137 }
